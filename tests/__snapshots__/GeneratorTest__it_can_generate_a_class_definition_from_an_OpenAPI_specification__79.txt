<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Actions;

use BombenProdukt\Conformist\Contracts\Connector;
use BombenProdukt\Conformist\Enums\HttpMethod;
use BombenProdukt\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/actions#add-selected-repository-to-an-organization-secret
 */
final class AddSelectedRepoToOrgSecretRequest extends AbstractRequest
{
    public function __constructor(
        private string $org,
        private string $secretName,
        private int $repositoryId,
    ) {
        //
    }

    public function path(): string
    {
        return '/orgs/{org}/actions/secrets/{secret_name}/repositories/{repository_id}';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::PUT;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'org' => $this->org,
                'secret_name' => $this->secretName,
                'repository_id' => $this->repositoryId,
            ]),
        ];
    }
}
