<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Reactions;

use BombenProdukt\Conformist\Contracts\Connector;
use BombenProdukt\Conformist\Enums\HttpMethod;
use BombenProdukt\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/reactions#delete-an-issue-reaction
 */
final class DeleteForIssueRequest extends AbstractRequest
{
    public function __constructor(
        private string $owner,
        private string $repo,
        private int $issueNumber,
        private int $reactionId,
    ) {
        //
    }

    public function path(): string
    {
        return '/repos/{owner}/{repo}/issues/{issue_number}/reactions/{reaction_id}';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::DELETE;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'owner' => $this->owner,
                'repo' => $this->repo,
                'issue_number' => $this->issueNumber,
                'reaction_id' => $this->reactionId,
            ]),
        ];
    }
}
