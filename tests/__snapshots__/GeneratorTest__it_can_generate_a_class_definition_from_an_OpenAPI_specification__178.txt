<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Actions;

use BombenProdukt\Conformist\Contracts\Connector;
use BombenProdukt\Conformist\Enums\HttpMethod;
use BombenProdukt\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/actions#get-repository-required-workflow-usage
 */
final class GetRepoRequiredWorkflowUsageRequest extends AbstractRequest
{
    public function __constructor(
        private string $org,
        private string $repo,
        private int $requiredWorkflowIdForRepo,
    ) {
        //
    }

    public function path(): string
    {
        return '/repos/{org}/{repo}/actions/required_workflows/{required_workflow_id_for_repo}/timing';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::GET;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'org' => $this->org,
                'repo' => $this->repo,
                'required_workflow_id_for_repo' => $this->requiredWorkflowIdForRepo,
            ]),
        ];
    }
}
