<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Codespaces;

use BombenProdukt\Conformist\Contracts\Connector;
use BombenProdukt\Conformist\Enums\HttpMethod;
use BombenProdukt\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/codespaces#list-secrets-for-the-authenticated-user
 */
final class ListSecretsForAuthenticatedUserRequest extends AbstractRequest
{
    public function __constructor(
        private int $perPage,
        private int $page,
    ) {
        //
    }

    public function path(): string
    {
        return '/user/codespaces/secrets';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::GET;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithQueryParameters([
                'per_page' => $this->perPage,
                'page' => $this->page,
            ]),
        ];
    }
}
